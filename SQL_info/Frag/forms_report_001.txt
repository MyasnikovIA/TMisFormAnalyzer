              begin

                -- отделения на которые есть права
                select tb.COLUMN_VALUE
                  bulk collect into :ACCESSIBLE_DEPS
                  from table (D_PKG_CSE_ACCESSES.GET_ID_WITH_RIGHTS(pnLPU      => to_number(:LPU),
                                                                    psUNITCODE => 'DEPS',
                                                                    psRIGHT    => :DEP_RIGHT,
                                                                    pnCABLAB   => to_number(:CABLAB))) tb
                 union all
                select -1
                  from dual;
      
                -- Этот кусок нужно будет убрать, когда все ЮФ откажутся от RDEPS в пользу ACCESSIBLE_DEPS
                select D_STRAGG(tb.COLUMN_VALUE)
                  into :RDEPS
                  from table(cast(:ACCESSIBLE_DEPS as D_CL_ID)) tb;
                :RDEPS := ';' || :RDEPS || ';';
                select dp.ID
                  bulk collect into :REANIMATION_DEPS
                  from D_V_DEPS_TYPES dtt
                       join D_V_DEPS_BASE dp on dp.DP_TYPE_ID = dtt.ID
                 where dp.LPU = to_number(:LPU)
                   and dtt.DT_CODE = '5'
                   and dtt.VERSION = (select D_PKG_VERSIONS.GET_VERSION_BY_LPU(0, to_number(:LPU), 'DEPS_TYPES')
                                        from dual)
                 union all
                select -1
                  from dual;
      
                -- права на HH_DEP для SubForm
                D_PKG_URPRIVS.GET_STANDART_PRIVS(pnLPU      => :LPU,
                                                 psUNITCODE => 'HOSP_HISTORY_DEPS',
                                                 pnCID      => null,
                                                 pnINSERT   => nT,
