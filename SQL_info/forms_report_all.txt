  [14] M2 Action: checkBeforeRegForHosp
      Источник: /var/www/t-mis/mis/Forms/HospPlan/hp_directions_edit.frm
      SQL:
      
              /**
               * проверяем использование направленния
               * направление по связи с D_OUTER_DIRECTION.REPRESENT_DIRECTION
               * или направление D_DIRECTIONS (в случае внутреннего направления)
               * получаем из SelectAction
               */
              begin
                :CURRENT_DIR_ID := to_number(coalesce(:OUTER_ID, :DIR_ID));
                begin
                  :SHC_REG_EXIST := 1;
                  select (select count(*)
                            from D_V_DIRECTION_SERVICES_BASE ds
                           where ds.DISEASECASE = hpj.DISEASECASE
                             and ds.SERV_STATUS != 0),
                         (select count(*)
                            from D_V_HOSP_HISTORIES_BASE hh
                           where hh.HPK_PLAN_JOURNAL = hpj.ID),
                         hpk.HP_NAME,
                         to_char(sr.DATE_REC, 'DD.MM.YYYY HH24:MI')
                    into :SERV_NO_HH,
                         :HH_EXIST,
                         :HP_NAME,
                         :JOUR_REC_DATE
                    from D_V_DIRECTIONS_BASE dir
                         join D_V_HPK_SCHEDULE_REG sr on sr.DIRECTION = dir.ID
                         join D_V_HPK_PLAN_JOURNALS_BASE hpj on hpj.DIRECTION = dir.ID
                         join D_V_HOSP_PLAN_KINDS hpk on hpj.HPK = hpk.ID
                   where dir.ID = :CURRENT_DIR_ID;
                exception when no_data_found then
                  :SHC_REG_EXIST := 0;
                  :SERV_NO_HH := 0;
                  :HH_EXIST := 0;
                  :HP_NAME := null;
                  :JOUR_REC_DATE := null;
                end;
                if :SERV_NO_HH = 0 and :HH_EXIST = 0 then
                  :HPK_EXIST := 1;
                  begin
                    select hpk.HP_NAME,
                           coalesce(:JOUR_REC_DATE, to_char(hp.PLAN_DATE, 'DD.MM.YYYY HH24:MI'))
                      into :HP_NAME,
                           :JOUR_REC_DATE
                      from D_V_HPK_PLAN_JOURNALS_BASE hpj
                           join D_V_HOSP_PLAN_KINDS hpk on hpj.HPK = hpk.ID
                           join D_V_HPK_PLANS_BASE hp on hpj.HPK_PLAN = hp.ID
                     where hpj.DIRECTION = :CURRENT_DIR_ID;
                  exception when no_data_found then
                    :HPK_EXIST := 0;
                    :HP_NAME := null;
                    :JOUR_REC_DATE := null;
                  end;
                end if;
              end;
